namespace CsharpExercicios;

class Program {
    static void Main(string[] args) {
        ApresentarPrograma();
        Calculo();
    }

    public static void ApresentarPrograma() {
        Console.WriteLine("Programa da Área dos Retângulos");
    }

    public static void Calculo() {
        Console.Write("Altura 1: ");
        double a1 = Convert.ToDouble(Console.ReadLine());

        Console.Write("Largura 1: ");
        double l1 = Convert.ToDouble(Console.ReadLine());

        Console.Write("Altura 2: ");
        double a2 = Convert.ToDouble(Console.ReadLine());

        Console.Write("Largura 2: ");
        double l2 = Convert.ToDouble(Console.ReadLine());

        double area1 = a1 * l1;
        double area2 = a2 * l2;

        bool iguais = false;

        if (area1 == area2) {
            iguais = true;
        }

        Console.WriteLine($"O primeiro retângulo possui área {area1}.");
        Console.WriteLine($"O primeiro retângulo possui área {area2}.");
        Console.WriteLine($"Eles são iguais? {iguais}");
    }
}
